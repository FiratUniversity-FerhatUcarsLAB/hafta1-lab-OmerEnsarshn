ALGORİTMA: YüksekSkorAlma

BAŞLA
    // Başlangıç ayarları
    skor ← 0
    can ← 3
    seviye ← 1

    OYUNU_BAŞLAT()

    // Oyun devam ettiği sürece
    WHILE can > 0 DO
        hareket ← KULLANICI_HAREKETİ_AL()

        // Oyuncunun yaptığı harekete göre sonuç
        IF hareket == "doğru hamle" THEN
            skor ← skor + (10 * seviye)
            YAZDIR("Tebrikler! Skorun: ", skor)

            // Belirli bir skora ulaşınca seviye atla
            IF skor ≥ (seviye * 100) THEN
                seviye ← seviye + 1
                YAZDIR("Seviye atladın! Yeni seviye: ", seviye)
            ENDIF

        ELSE IF hareket == "hatalı hamle" THEN
            can ← can - 1
            YAZDIR("Can kaybettin! Kalan can: ", can)
        ELSE
            YAZDIR("Geçersiz hareket!")
        ENDIF

        // Bonus durum kontrolü
        IF skor % 200 == 0 AND skor != 0 THEN
            can ← can + 1
            YAZDIR("Bonus can kazandın! Yeni can: ", can)
        ENDIF
    ENDWHILE

    YAZDIR("Oyun bitti! Toplam skorun: ", skor)

    // Skor kontrolü
    IF skor ≥ 1000 THEN
        YAZDIR("Mükemmel performans! En yüksek skor senin!")
    ELSE
        YAZDIR("Daha fazla pratik yapmalısın.")
    ENDIF
BİTİR
