Lütfen seçtiğiniz algoritmaya ait çözümü ve diğer isterleri aşağıya ekleyiniz:

ALGORİTMA: BasitKartOyunuStratejisi

BAŞLA
    // Oyuncu ve rakip başlangıç kartları
    oyuncu_kartlar ← 5
    rakip_kartlar ← 5
    skor ← 0
    tur ← 1

    OYUNU_BAŞLAT()

    WHILE oyuncu_kartlar > 0 AND rakip_kartlar > 0 DO
        YAZDIR("Tur: ", tur)
        oyuncu_hamle ← KULLANICI_HAMLE_AL()
        rakip_hamle ← RAKIP_HAMLE_OLUSTUR()

        // Hamle karşılaştırması
        IF oyuncu_hamle > rakip_hamle THEN
            skor ← skor + 1
            YAZDIR("Tur kazanıldı! Skor: ", skor)
        ELSE IF oyuncu_hamle < rakip_hamle THEN
            skor ← skor - 1
            YAZDIR("Tur kaybedildi! Skor: ", skor)
        ELSE
            YAZDIR("Berabere tur!")
        ENDIF

        // Kartları azalt
        oyuncu_kartlar ← oyuncu_kartlar - 1
        rakip_kartlar ← rakip_kartlar - 1
        tur ← tur + 1

        // Risk değerlendirme
        IF skor < -2 THEN
            YAZDIR("Risk yüksek! Daha dikkatli hamle yap")
        ENDIF
    ENDWHILE

    // Oyun bitişi
    IF skor > 0 THEN
        YAZDIR("Oyun kazanıldı! Toplam skor: ", skor)
    ELSE IF skor < 0 THEN
        YAZDIR("Oyun kaybedildi. Toplam skor: ", skor)
    ELSE
        YAZDIR("Oyun berabere. Toplam skor: ", skor)
    ENDIF
BİTİR
